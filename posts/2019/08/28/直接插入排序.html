# 实现原理

​ 将数组分为两部分，左边为有序集合，右边为无序集合，取右边集合的第一个元素与左边的集合挨个对比，直到插入合适的位置后退出比较

# 实现

inputArr = [10, 34, 29, 4, 0, 34, 5, 1, 8]

length = len(inputArr)
times = 0
print("未排序集合:{0}".format(inputArr))
for maxIndex in range(1, length):
    for minIndex in range(0, maxIndex-1):
        maxItem = inputArr[maxIndex]
        minItem = inputArr[minIndex]
        if(minItem > maxItem):
            inputArr[minIndex], inputArr[maxIndex] = maxItem, minItem
        times += 1
        print("第{0}次排序".format(times))
        print("  left:{0}".format(inputArr[0:maxIndex]))
        print("  right:{0}".format(inputArr[maxIndex:length]))

        pass
print("已排序集合:{0}".format(inputArr))
